Multi-Threading
( Not supported by Internet Explorer )

Function vkthread.execAll() opens a door for implementation a concept of "parallel computing".
This function let you to start simultaneous execution of multiple functions in parallel threads. On multi-core CPU it can significantly increase application performance. 

function normalize(arr){
	return arr.map(function(elm){
		return elm.trim().toLowerCase();
	});
} 



var arr = ['ONE ', ' TWo ', 'ThReE', 'FouR ', ' Five ', 'sIX', '      SEVEn'],
    param1 = {
	    fn: normalize,
	    args: [arr.slice(0,3)]
    },
    param2 = {
	    fn: normalize,
	    args: [arr.slice(3, arr.length)]
    };

vkthread.execAll([param1,param2]).then(
    function (data) {
        console.log(data[0].concat(data[1]);
    }
);
