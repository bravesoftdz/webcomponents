Lambda Function

Lambda. Have fun! 


function greetLambda(arg) {
    var displayMessage = (function(msg1){
        return (msg2) => msg1 + msg2; // ES6 Arrow function
    }(arg));
    return displayMessage("Lambda World!");
}




/* ------------ Promise Style ------------ */

var param = {
    fn: greetLambda,
    args: ['Hello, ']
}

vkthread.exec(param).then(
    function(data){
	    console.log(data);
    }
);

/* ------------ Callback Style ------------ */

function print(data){
    document.getElementById('demo_result_thread').innerHTML = data;
}

var param = {
    fn: greetLambda,
    args: ['Hello, '],
    cb:print
}

vkthread.exec(param);
