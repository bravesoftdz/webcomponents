Anonymous Function

Anonymous function also can be run in a thread. It can also be ES6 Arrow function.





/* ------- Promise style ------- */

var param = {
	fn: function(arr){
            return arr.map(Math.sqrt)
        },
	args: [[1, 4, 9]]
}

vkthread.exec(param).then(
	function(data){
		console.log(data);
	}
);

/* ------- Callback style ------- */

function display(data){ // <-- callback function
	console.log(data);
}

var param = {
	fn: function(arr){
            return arr.map(Math.sqrt)
        },
	args: [[1, 4, 9]],
	cb: display    // <-- callback function
}

vkthread.exec(param);
