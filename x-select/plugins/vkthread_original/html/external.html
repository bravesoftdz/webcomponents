Remote Function

You can easily use remote function in a thread
In this case function makeNamesUpper( ) is located in external file.

projects/app/vkthread/js/my_utils.js 
. . .

function makeNamesUpper(arr){

   return arr.map(function(elm){
    	return elm.toUpperCase();
    });
}

. . .




/* ------------ Promise Style ------------ */

var param = {
    fn:'makeNamesUpper', // <-- function's name; must be a String
    args: [['one', 'two', 'three']],
    importFiles:['http://localhost/projects/app/vkthread/js/my_utils.js']
};

vkthread.exec(param).then(
   function (data) {
      console.log(data);
    }
);

/* ------------ Callback Style ------------ */

var param = {
      fn:'makeNamesUpper',
      args: [['one', 'two', 'three']],
      importFiles:['http://localhost/projects/app/vkthread/js/my_utils.js'],
      cb:function(data){
        console.log(data;
      }
   };

  vkthread.exec(param);
