Standalone Function

This page demonstrates advantage of using vkThread() plugin.

This function runs extensive process which takes a few seconds (on my comp), but UI is not affected. 

/* ------- Function to run in a thread  -------*/
function foo(str) {
    var ret, ix;
    for(ix=0;ix<1e7;ix++){
	    ret = str.split(',').join('-');
    }
    return ret;
}




/* ------------ Promise Style ------------ */
var param = {
	fn: foo,
	args: ['a,b,c']
};

vkthread.exec(param).then(function(data){
	console.log(data);
});

/* ------------ Callback Style ------------ */

function fooCb (data) { // <-- callback function
	console.log(data.toUpperCase());
}

var param = {
	fn: foo,
	args: ['a,b,c'],
	cb:fooCb // <-- callback function can be anonymouse;

};

vkthread.exec(param);
